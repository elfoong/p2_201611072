@startuml
start
:user select number;
:trycount=0;
repeat
:user guess number;
if (select number>guess number)
:print "Up";
:trycount=trycount+1;
elseif (select number<guess number)
:print "Down";
:trycount=trycount+1;
endif
repeat while(select number==guess number)
:print "You win at 'trycount' try!";
stop
@enduml
